**vue3.0 子组件调用父组件的方法**

<!-- more -->

##### 父组件

```
<div>
  <register @update:changeDisplayValue="changeDisplayValue"></register>
</div>

setup(){
  let changeDisplayValue = value => {
    console.log(value);
    loginData.displayValue = value;
  };
  return {
    changeDisplayValue
  };
}
```

##### 子组件

```
<template>
  <div class="register">
    <h3>注册</h3>
    <span @click="toLogin">去登录</span>
  </div>
</template>

<script>
export default {
  setup(props, context) {
    let toLogin = () => {
      context.emit("update:changeDisplayValue", "login");
    };
    return {
      toLogin
    };
  }
};
</script>
```
